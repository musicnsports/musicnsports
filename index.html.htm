    <script>

        // Meditation Timer App

        class MeditationTimer {

            constructor() {

                this.duration = 10 * 60; // seconds

                this.remaining = this.duration;

                this.interval = 5 * 60; // seconds

                this.isRunning = false;

                this.isPaused = false;

                this.timerId = null;

                this.startTime = null;

                this.pausedTime = 0;

                this.sessionCount = parseInt(localStorage.getItem('sessionCount') || '0');

                this.totalMinutes = parseInt(localStorage.getItem('totalMinutes') || '0');

               

                this.initElements();

                this.initEventListeners();

                this.updateStats();

                this.updateDisplay();

            }

           

            initElements() {

                this.timeDisplay = document.getElementById('timeDisplay');

                this.phaseDisplay = document.getElementById('phaseDisplay');

                this.startBtn = document.getElementById('startBtn');

                this.pauseBtn = document.getElementById('pauseBtn');

                this.resetBtn = document.getElementById('resetBtn');

                this.durationSlider = document.getElementById('durationSlider');

                this.intervalSlider = document.getElementById('intervalSlider');

                this.durationValue = document.getElementById('durationValue');

                this.intervalValue = document.getElementById('intervalValue');

                this.sessionCountEl = document.getElementById('sessionCount');

                this.totalMinutesEl = document.getElementById('totalMinutes');

                this.progressCircle = document.querySelector('.progress-circle');

                this.presetBtns = document.querySelectorAll('.preset-btn');

                this.container = document.querySelector('.container');

            }

           

            initEventListeners() {

                this.startBtn.addEventListener('click', () => this.start());

                this.pauseBtn.addEventListener('click', () => this.pause());

                this.resetBtn.addEventListener('click', () => this.reset());

               

                this.durationSlider.addEventListener('input', (e) => {

                    this.duration = parseInt(e.target.value) * 60;

                    if (!this.isRunning) {

                        this.remaining = this.duration;

                        this.updateDisplay();

                    }

                    this.durationValue.textContent = e.target.value + ' min';

                    this.updateActivePreset();

                });

               

                this.intervalSlider.addEventListener('input', (e) => {

                    this.interval = parseInt(e.target.value) * 60;

                    this.intervalValue.textContent = e.target.value + ' min';

                });

               

                this.presetBtns.forEach(btn => {

                    btn.addEventListener('click', () => {

                        const minutes = parseInt(btn.dataset.minutes);

                        this.durationSlider.value = minutes;

                        this.duration = minutes * 60;

                        if (!this.isRunning) {

                            this.remaining = this.duration;

                            this.updateDisplay();

                        }

                        this.durationValue.textContent = minutes + ' min';

                        this.updateActivePreset();

                    });

                });

            }

           

            updateActivePreset() {

                const minutes = this.duration / 60;

                this.presetBtns.forEach(btn => {

                    if (parseInt(btn.dataset.minutes) === minutes) {

                        btn.classList.add('active');

                    } else {

                        btn.classList.remove('active');

                    }

                });

            }

           

            start() {

                if (this.isPaused) {

                    this.isPaused = false;

                    this.startTime = Date.now() - (this.duration - this.remaining) * 1000;

                } else {

                    this.startTime = Date.now();

                    this.remaining = this.duration;

                    this.container.classList.add('breathing');

                }

               

                this.isRunning = true;

                this.startBtn.disabled = true;

                this.pauseBtn.disabled = false;

                this.phaseDisplay.textContent = 'Meditating';

               

                this.timerId = setInterval(() => this.tick(), 100);

                this.playStartSound();

            }

           

            pause() {

                this.isPaused = true;

                this.isRunning = false;

                clearInterval(this.timerId);

                this.startBtn.disabled = false;

                this.pauseBtn.disabled = true;

                this.phaseDisplay.textContent = 'Paused';

                this.container.classList.remove('breathing');

            }

           

            reset() {

                this.isRunning = false;

                this.isPaused = false;

                clearInterval(this.timerId);

                this.remaining = this.duration;

                this.startBtn.disabled = false;

                this.pauseBtn.disabled = true;

                this.phaseDisplay.textContent = 'Ready';

                this.container.classList.remove('breathing');

                this.updateDisplay();

            }

           

            tick() {

                const elapsed = Math.floor((Date.now() - this.startTime) / 1000);

                this.remaining = Math.max(0, this.duration - elapsed);

               

                // Check for interval bells

                const timePassed = this.duration - this.remaining;

                if (timePassed > 0 && timePassed % this.interval === 0 && this.remaining > 0) {

                    this.playIntervalSound();

                }

               

                this.updateDisplay();

               

                if (this.remaining <= 0) {

                    this.complete();

                }

            }

           

            complete() {

                this.isRunning = false;

                clearInterval(this.timerId);

                this.container.classList.remove('breathing');

                this.phaseDisplay.textContent = 'Complete';

                this.startBtn.disabled = false;

                this.pauseBtn.disabled = true;

               

                // Update stats

                this.sessionCount++;

                this.totalMinutes += Math.floor(this.duration / 60);

                localStorage.setItem('sessionCount', this.sessionCount);

                localStorage.setItem('totalMinutes', this.totalMinutes);

                this.updateStats();

               

                this.playCompleteSound();

               

                setTimeout(() => {

                    this.reset();

                }, 3000);

            }

           

            updateDisplay() {

                const minutes = Math.floor(this.remaining / 60);

                const seconds = this.remaining % 60;

                this.timeDisplay.textContent =

                    String(minutes).padStart(2, '0') + ':' +

                    String(seconds).padStart(2, '0');

               

                // Update progress circle

                const progress = 1 - (this.remaining / this.duration);

                const circumference = 2 * Math.PI * 90;

                const offset = circumference - (progress * circumference);

                this.progressCircle.style.strokeDashoffset = offset;

            }

           

            updateStats() {

                this.sessionCountEl.textContent = this.sessionCount;

                this.totalMinutesEl.textContent = this.totalMinutes;

            }

           

            playStartSound() {

                this.playTone(440, 0.2, 'sine');

            }

           

            playIntervalSound() {

                this.playTone(523.25, 0.3, 'sine');

                setTimeout(() => this.playTone(659.25, 0.3, 'sine'), 200);

            }

           

            playCompleteSound() {

                this.playTone(523.25, 0.2, 'sine');

                setTimeout(() => this.playTone(659.25, 0.2, 'sine'), 200);

                setTimeout(() => this.playTone(783.99, 0.4, 'sine'), 400);

            }

           

            playTone(frequency, duration, type = 'sine') {

                try {

                    const audioContext = new (window.AudioContext || window.webkitAudioContext)();

                    const oscillator = audioContext.createOscillator();

                    const gainNode = audioContext.createGain();

                   

                    oscillator.connect(gainNode);

                    gainNode.connect(audioContext.destination);

                   

                    oscillator.type = type;

                    oscillator.frequency.value = frequency;

                   

                    gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);

                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + duration);

                   

                    oscillator.start(audioContext.currentTime);

                    oscillator.stop(audioContext.currentTime + duration);

                } catch (e) {

                    console.log('Audio not supported');

                }

            }

        }

       

        // Initialize app when DOM is ready

        document.addEventListener('DOMContentLoaded', () => {

            new MeditationTimer();

        });

    </script>
