Developer Guide: Upgrade to Next.js + Supabase + Vercel

1. Deploy this file now - free on Vercel/Netlify

Download this HTML file, push to GitHub, import to Vercel or drag-drop to Netlify. It works fully with localStorage today. Zero build step.

2. Upgrade to Next.js + Supabase in 10 minutes

npx create-next-app@latest musicnsports
cd musicnsports
npm install @supabase/supabase-js

3. Supabase SQL: profiles/videos tables

-- Enable Row Level Security
CREATE TABLE profiles (
  id UUID REFERENCES auth.users PRIMARY KEY,
  username TEXT UNIQUE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE videos (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users,
  title TEXT NOT NULL,
  description TEXT,
  video_url TEXT NOT NULL,
  video_id TEXT NOT NULL,
  platform TEXT NOT NULL CHECK (platform IN ('youtube', 'vimeo')),
  thumbnail TEXT,
  tags TEXT[],
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- RLS Policies
ALTER TABLE videos ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Videos are viewable by everyone" 
ON videos FOR SELECT USING (true);

CREATE POLICY "Users can insert their own videos" 
ON videos FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can update own videos" 
ON videos FOR UPDATE USING (auth.uid() = user_id);

4. parseVideoUrl function - reuse from this file

// utils/parseVideoUrl.ts
export function parseVideoUrl(url: string) {
  const ytRegex = /(?:youtube\.com\/(?:watch\?v=|embed\/|v\/)|youtu\.be\/)([\w-]{11})/;
  const vimeoRegex = /vimeo\.com\/(\d+)/;
  
  const ytMatch = url.match(ytRegex);
  if (ytMatch) return {
    platform: 'youtube',
    id: ytMatch[1],
    thumbnail: `https://img.youtube.com/vi/${ytMatch[1]}/hqdefault.jpg`,
    embed: `https://www.youtube.com/embed/${ytMatch[1]}`
  };
  
  const vimeoMatch = url.match(vimeoRegex);
  if (vimeoMatch) return {
    platform: 'vimeo',
    id: vimeoMatch[1],
    thumbnail: '/vimeo-placeholder.jpg',
    embed: `https://player.vimeo.com/video/${vimeoMatch[1]}`
  };
  
  return null;
}

5. Swap localStorage for Supabase

Replace the `storage` object in this file with Supabase client calls:

// lib/supabase.ts
import { createClient } from '@supabase/supabase-js'
export const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
)

// Replace storage.getVideos() with:
const { data } = await supabase.from('videos')
  .select('*').order('created_at', { ascending: false })

// Replace storage.addVideo() with:
const parsed = parseVideoUrl(url)
await supabase.from('videos').insert({
  user_id: user.id,
  title, description, video_url: url,
  video_id: parsed.id, platform: parsed.platform,
  thumbnail: parsed.thumbnail, tags
})
